\subsection{1.4 Dat cau hoi phan tich}

\begin{frame}[t]{1.4 Dat Cau Hoi Phan Tich}
\begin{block}{Muc tieu cua buoc 1.4}
Xac dinh pham vi giai phap va cac rang buoc ky thuat cho du an VietVibe.
\end{block}

\begin{itemize}
    \item Cac cau hoi duoc rut ra truc tiep tu P1--P4.
    \item Moi cau hoi dan den mot yeu cau he thong co the do luong.
    \item Ket qua la co so de xac dinh Input/Output va kien truc MAS.
\end{itemize}
\end{frame}

\begin{frame}[t]{Nhom Cau Hoi Cot Loi (1/2)}
\begin{enumerate}
    \item Tich hop du lieu da nguon nhu the nao?
    \item Bo sung chieu sau van hoa tu dong bang cach nao?
    \item Dich theo ngu canh van hoa khac gi dich ngon ngu thong thuong?
\end{enumerate}

\vspace{0.4cm}
\begin{alertblock}{Y nghia}
Ba cau hoi dau dinh hinh Data Adapters, Enrichment Pipeline va Cultural Translation Agent.
\end{alertblock}
\end{frame}

\begin{frame}[t]{Nhom Cau Hoi Cot Loi (2/2)}
\begin{enumerate}
    \setcounter{enumi}{3}
    \item Lam sao giu du lieu luon moi khi nguon thay doi lien tuc?
    \item Lam sao ca nhan hoa trai nghiem theo vi tri, so thich, ngon ngu, thoi diem?
    \item He thong co kha thi voi gan nhu zero-cost (free tier) hay khong?
\end{enumerate}

\vspace{0.4cm}
\begin{block}{Tac dong thiet ke}
Dinh nghia co che scheduler, ranking context-aware, va cac quyet dinh toi uu chi phi ngay tu dau.
\end{block}
\end{frame}

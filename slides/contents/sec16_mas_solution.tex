\subsection{1.6 Giai phap de xuat - MAS}

\begin{frame}[t]{1.6 Giai Phap De Xuat: VietVibe MAS}
\begin{block}{Vi sao Multi-Agent System?}
Bai toan gom nhieu loai xu ly khac nhau (crawl, enrich, translate, monitor, rank) nen can tach thanh cac don vi tu chu.
\end{block}

\begin{itemize}
    \item De phat trien song song va nang cap tung phan doc lap.
    \item De giam coupling giua crawl logic va AI logic.
    \item De de quan sat, retry, va bao tri theo tung agent.
\end{itemize}
\end{frame}
